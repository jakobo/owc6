Title: A Pragmatic Guide to Web Components

Overview: Web Components. The moment is upon us where we can encapsulate large amounts of code behind a single semantic tag. Get a crash course on what Web Components are, how the current batch of JavaScript frameworks address the technology, and some good anti-patterns to avoid when building your own Web Components.

        App = Ember.Application.create();
        App.IndexRoute = Ember.Route.extend({});
        App.ECarouselComponent = Ember.Component.extend({
          items: Ember.A([]),
          index: 0,
          didInsertElement: function() {
            this._super();
            var self = this;
            var $items = this.$('content carousel-item');
            $items.hide();
            $items.each(function(idx, item) {
              self.items.pushObject(Ember.Object.create({
                className: '',
                content: self.$(item).html()
              }));
            });
            $items.remove();
            this.go(0);
          },

            next: function() { this.go(this.get('index') + 1); }
          },
          go: function(idx) {
            if (idx >= this.items.length) idx = 0;
            if (idx < 0) idx = this.items.length - 1;
            this.items.objectAt(this.get('index')).set('className', '');
            this.items.objectAt(idx).set('className', 'selected');
            this.set('index', idx);
          }
        });

              $scope.items = [];

                var div = angular.element(document.createElement('div'));

          /* ... */
          attached: function() {
            var self = this,
                contents = this.$.content.getDistributedNodes();
            this.items = [];
            this.index = 0;

              this.items.push({

            this.$.prev.addEventListener('click', function() {
              self.go(self.index - 1);
            });
            this.$.next.addEventListener('click', function() {
              self.go(self.index + 1);
            });
            this.go(0);
